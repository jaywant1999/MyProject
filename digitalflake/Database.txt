create database digitalflake;


create table category (
ID int(10),
Name varchar(50),
Description varchar(100),
Status Enum('active','inactive'));

create table Products (
ID int(100) primary key AUTO_INCREMENT,
Name varchar(100),
PackSize varchar(100),
Category varchar(100),
MRP int(100),
image BLOB,
Status Enum('active','inactive'));

create table Login (
username varchar(50),
password varchar(15));
